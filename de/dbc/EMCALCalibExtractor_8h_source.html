60  template <typename... axes>
61  o2::emcal::BadChannelMap calibrateBadChannels(boost::histogram::histogram<axes...>& hist)
62  {
63  // calculate the mean and sigma of the mEsumHisto
64  auto fitValues = o2::utils::fitBoostHistoWithGaus<double>(hist);
65  double mean = fitValues.at(1);
66  double sigma = fitValues.at(2);
67 
68  // calculate the "good cell window from the mean"
69  double maxVal = mean + mSigma * sigma;
70  double minVal = mean - mSigma * sigma;
71  o2::emcal::BadChannelMap mOutputBCM;
72  // now loop through the cells and determine the mask for a given cell
73  for (int cellID = 0; cellID < 17664; cellID++) {
74  double E = hist.at(cellID);
75  // if in the good window, mark the cell as good
76  // for now we won't do warm cells - the definition of this is unclear.
77  if (E == 0) {
78  mOutputBCM.addBadChannel(cellID, o2::emcal::BadChannelMap::MaskType_t::DEAD_CELL);
79  } else if (E < maxVal || E > minVal) {
80  mOutputBCM.addBadChannel(cellID, o2::emcal::BadChannelMap::MaskType_t::GOOD_CELL);
81  } else {
82  mOutputBCM.addBadChannel(cellID, o2::emcal::BadChannelMap::MaskType_t::BAD_CELL);
83  }
84  }
85 
86  return mOutputBCM;
87  }
