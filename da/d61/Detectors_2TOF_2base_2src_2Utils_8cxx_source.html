55 double Utils::subtractInteractionBC(double time, bool subLatency)
56 {
57  static const int deltalat = o2::tof::Geo::BC_IN_ORBIT - o2::tof::Geo::LATENCYWINDOW_IN_BC;
58  int bc = int(time * o2::tof::Geo::BC_TIME_INPS_INV + 0.2);
59 
60  if (subLatency) {
61  if (bc >= o2::tof::Geo::LATENCYWINDOW_IN_BC) {
62  bc -= o2::tof::Geo::LATENCYWINDOW_IN_BC;
63  time -= o2::tof::Geo::LATENCYWINDOW_IN_BC * o2::tof::Geo::BC_TIME_INPS;
64  } else {
65  bc += deltalat;
66  time += deltalat * o2::tof::Geo::BC_TIME_INPS;
67  }
68  }
69 
70  int bcOrbit = bc % o2::constants::lhc::LHCMaxBunches;
71 
72  int dbc = o2::constants::lhc::LHCMaxBunches, bcc = bc;
73  for (int k = 0; k < getNinteractionBC(); k++) { // get bc from fill scheme closest
74  if (abs(bcOrbit - getInteractionBC(k)) < dbc) {
75  bcc = bc - bcOrbit + getInteractionBC(k);
76  dbc = abs(bcOrbit - getInteractionBC(k));
77  }
78  if (abs(bcOrbit - getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches) < dbc) { // in case k is close to the right border (last BC of the orbit)
79  bcc = bc - bcOrbit + getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches;
80  dbc = abs(bcOrbit - getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches);
81  }
82  if (abs(bcOrbit - getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches) < dbc) { // in case k is close to the left border (BC=0)
83  bcc = bc - bcOrbit + getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches;
84  dbc = abs(bcOrbit - getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches);
85  }
86  }
87  time -= o2::tof::Geo::BC_TIME_INPS * bcc;
88 
89  return time;
90 }

129 void Utils::addBC(float toftime, bool subLatency)
130 {
131  if (!mIsInit) {
132  init();
133  mIsInit = true;
134  }
135 
136  if (mNautodet > MAX_NUM_EVENT_AUTODETECT) {
137  if (!hasFillScheme()) { // detect fill scheme
138  int thres = mMaxBC / 2;
139  for (int i = 0; i < o2::constants::lhc::LHCMaxBunches; i++) {
140  if (mBCmult[i] > thres) { // good bunch
141  addInteractionBC(i);
142  }
143  }
144  }
145  return;
146  }
147 
148  // just fill
149  static const int deltalat = o2::tof::Geo::BC_IN_ORBIT - o2::tof::Geo::LATENCYWINDOW_IN_BC;
150  int bc = int(toftime * o2::tof::Geo::BC_TIME_INPS_INV + 0.2) % o2::constants::lhc::LHCMaxBunches;
151 
152  if (subLatency) {
153  if (bc >= o2::tof::Geo::LATENCYWINDOW_IN_BC) {
154  bc -= o2::tof::Geo::LATENCYWINDOW_IN_BC;
155  toftime -= o2::tof::Geo::LATENCYWINDOW_IN_BC * o2::tof::Geo::BC_TIME_INPS;
156  } else {
157  bc += deltalat;
158  toftime += deltalat * o2::tof::Geo::BC_TIME_INPS;
159  }
160  }
161 
162  mBCmult[bc]++;
163 
164  if (mBCmult[bc] > mMaxBC) {
165  mMaxBC = mBCmult[bc];
166  }
167  mNautodet++;
168 }
